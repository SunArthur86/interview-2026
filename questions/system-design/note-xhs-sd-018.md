@Service
public class AntiFraudService {

@Scheduled(fixedRate = 60000)  // жҜҸеҲҶй’ҹжү«жҸҸ
public void releaseExpiredGroups() {
    // иҺ·еҸ–еҲҶеёғејҸй”ҒпјҢйҳІжӯўеӨҡе®һдҫӢйҮҚеӨҚеӨ„зҗҶ
    String lockKey = "lock:expire:scan";
    String lockValue = UUID.randomUUID().toString();
    
    Boolean locked = redis.opsForValue()
        .setIfAbsent(lockKey, lockValue, 55, TimeUnit.SECONDS);
    if (Boolean.FALSE.equals(locked)) return;
    
    try {
        // жҹҘжүҫ24е°Ҹж—¶еүҚеҲӣе»әдё”зҠ¶жҖҒдёә"еҫ…жҲҗеӣў"зҡ„жӢјеӣў
        List<Group> expired = groupRepository.findExpiredGroups(
            GroupStatus.PENDING, LocalDateTime.now().minusHours(24));
        
        for (Group group : expired) {
            // йҮҠж”ҫиҜҘеӣўжүҖжңүй”Ғе®ҡзҡ„еә“еӯҳ
            for (GroupMember member : group.getMembers()) {
                String lockDetailKey = "locked:detail:" + group.getId() + ":" + member.getUserId();
                // еӣһж»ҡпјҡеҸҜз”Ёеә“еӯҳ+1пјҢй”Ғе®ҡеә“еӯҳ-1
                redis.opsForValue().increment("stock:product:" + group.getProductId());
                redis.opsForValue().decrement("locked:product:" + group.getProductId());
                redis.delete(lockDetailKey);
            }
            // жӣҙж–°жӢјеӣўзҠ¶жҖҒ
            group.setStatus(GroupStatus.FAILED);
            groupRepository.save(group);
            
            // йҖҡзҹҘеӣўе‘ҳ
            notifyService.notifyGroupFailed(group);
        }
    } finally {
        // йҮҠж”ҫй”ҒпјҲLuaдҝқиҜҒеҺҹеӯҗжҖ§пјү
        releaseLock(lockKey, lockValue);
    }
}

## вқ“ иӢҸж јжӢүеә•ејҸйқўиҜ•иҝҪй—®

1. **"дҪ иҜҙејҖеӣўж—¶й”Ғе®ҡеә“еӯҳпјҢдҪҶеҰӮжһңеҗҢдёҖе•Ҷе“ҒеҗҢж—¶жңү1000дёӘжӢјеӣўеңЁжҙ»еҠЁпјҢRedisзҡ„еә“еӯҳжҖҺд№Ҳз®ЎзҗҶпјҹжҳҜжҜҸдёӘеӣўзӢ¬з«ӢдёҖд»ҪиҝҳжҳҜе…ұдә«пјҹ"**
   вҶ’ е…ұдә«дёҖд»ҪеҸҜз”Ёеә“еӯҳжұ пјҢжҜҸдёӘеӣўй”Ғе®ҡзҡ„жҳҜжұ дёӯзҡ„д»ҪйўқгҖӮжҖ»ж•° = еҸҜз”Ё + жүҖжңүеӣўй”Ғе®ҡ + е·Іе”®еҮә

2. **"еҰӮжһңеӣўй•ҝејҖеӣўеҗҺпјҢеңЁ24е°Ҹж—¶еҶ…дёҖзӣҙйӮҖиҜ·дёҚеҲ°дәәпјҢдҪҶеә“еӯҳиў«й”Ғе®ҡдәҶеҜјиҮҙе…¶д»–з”ЁжҲ·д№°дёҚдәҶпјҢжҖҺд№Ҳе№іиЎЎпјҹ"**
   вҶ’ и®ҫзҪ®йҳ¶жўҜйҮҠж”ҫзӯ–з•ҘпјҡеүҚ2е°Ҹж—¶е…Ёй”ҒпјҢд№ӢеҗҺйҖҗжӯҘйҮҠж”ҫ30%/50%/е…ЁйғЁ

3. **"йҳІеҲ·и§„еҲҷеӨӘдёҘж јеҸҜиғҪиҜҜдјӨжӯЈеёёз”ЁжҲ·пјҢдҪ жҖҺд№Ҳе№іиЎЎпјҹ"**
   вҶ’ еҲҶзә§йЈҺжҺ§пјҡиҪ»зә§пјҲйӘҢиҜҒз ҒпјүвҶ’ дёӯзә§пјҲйҷҗжөҒйҷҚйҖҹпјүвҶ’ йҮҚзә§пјҲжӢҰжҲӘпјүгҖӮз»ҷз”ЁжҲ·з”іиҜүйҖҡйҒ“

4. **"жӢјеӣўдјҳжғ зҡ„еҲҶж‘Ҡз®—жі•пјҡеӣўй•ҝе…ҚеҚ•еңәжҷҜпјҢ3дәәеӣўпјҢеӣўй•ҝ0е…ғпјҢе…¶д»–2дәәе№іж‘ҠгҖӮйҖҖж¬ҫж—¶жҖҺд№ҲеӨ„зҗҶпјҹ"**
   вҶ’ жҢүе®һйҷ…ж”Ҝд»ҳжҜ”дҫӢйҖҖж¬ҫпјҢи®°еҪ•жҜҸдёӘжҲҗе‘ҳзҡ„е®һйҷ…ж”Ҝд»ҳйҮ‘йўқ

5. **"еҰӮжһңиҰҒж”ҜжҢҒдёҮдәәеӣўпјҲ10000дәәпјүпјҢжһ¶жһ„йңҖиҰҒжҖҺд№Ҳи°ғж•ҙпјҹ"**
   вҶ’ еҲҶзүҮзӯ–з•ҘпјҡжҢүеӣўIDе“ҲеёҢеҲ°дёҚеҗҢRedisе®һдҫӢпјҢйҒҝе…ҚеҚ•е®һдҫӢзғӯзӮ№

## з»“жһ„еҢ–еӣһзӯ”

**30 з§’з”өжўҜжј”и®Іпјҡ** жӢјеӣўжҠўиҙӯзі»з»ҹйңҖиҰҒи§ЈеҶідёүдёӘж ёеҝғй—®йўҳпјҡйҳІеҲ·пјҲз”ЁжҲ·з»ҙеәҰйҷҗжөҒ+и®ҫеӨҮжҢҮзә№+йЈҺжҺ§пјүгҖҒйҷҗжөҒпјҲзҪ‘е…іеұӮд»ӨзүҢжЎ¶+еә”з”ЁеұӮж»‘еҠЁзӘ—еҸЈпјүгҖҒеә“еӯҳдёҖиҮҙжҖ§пјҲRedisйў„жүЈ+DBд№җи§Ӯй”Ғ+и¶…ж—¶йҮҠж”ҫпјүгҖӮ

**еұ•ејҖжЎҶжһ¶пјҡ**
1. **жӢјеӣўеә“еӯҳжЁЎеһӢ** вҖ” еҸҜз”Ё = жҖ»йҮҸ - е·Ій”Ғе®ҡ - е·Іе”®еҮәпјҢдёҚжҳҜз®ҖеҚ•зҡ„DECR
2. **дёүеұӮйҳІжҠӨ** вҖ” йҳІеҲ·пјҲз”ЁжҲ·+и®ҫеӨҮ+IPйҷҗжөҒпјүвҶ’ йҷҗжөҒпјҲзҪ‘е…і+еә”з”ЁеҸҢеұӮпјүвҶ’ еә“еӯҳпјҲRedisйў„жүЈ+DBе…ңеә•пјү
3. **зҠ¶жҖҒжңә** вҖ” ејҖеӣў(й”Ғе®ҡеә“еӯҳ) вҶ’ еҫ…жҲҗеӣў вҶ’ жҲҗеӣў(жүЈеҮҸеә“еӯҳ) / и¶…ж—¶еӨұиҙҘ(йҮҠж”ҫеә“еӯҳ)

**ж”¶е°ҫпјҡ** иҝҷеқ—жҲ‘иё©иҝҮеқ‘вҖ”вҖ”иҰҒдёҚиҰҒж·ұе…ҘиҒҠпјҡжӢјеӣўи¶…ж—¶иҮӘеҠЁеӨұиҙҘзҡ„е®ҡж—¶д»»еҠЎпјҢеҰӮжһңд»»еҠЎжү§иЎҢж…ўеҜјиҮҙеә“еӯҳй•ҝж—¶й—ҙдёҚйҮҠж”ҫпјҢжҖҺд№ҲеҠһпјҹ
